// export default ProjectsPage;

// React Redux (connect)---------------
function mapStateToProps(state) {
  return {
    ...state.projectState,
  };
}

const mapDispatchToProps = {
  onLoad: loadProjects,
  onSave: saveProject,
};

export default connect(mapStateToProps, mapDispatchToProps)(ProjectsPage);
